matplotlib使用

Author: ww Date: 一月 9, 2018  11:08:28

matplotlib介绍

Matplotlib是一个基于NumPy阵列的多平台数据可视化库

1.导入Matplotlib

from matplotlib import pyplot as plt
import numpy as np

#设置样式
plt.style.use(‘classic’)

#线性

x = np.linspace(0,10,100)
fig = plt.figure()
plt.plot(x,np.sin(x),’-‘)
plt.plot(x,np.cos(x),’–’)

#保存图片

用来正常显示负号

plt.rcParams[‘axes.unicode_minus’]=False

中文显示不正常导入

import matplotlib

# plt.subplot(2,2,3) #指定画布的范围 2行2列 那个位置

#plot折线图

plt.xlabel(‘X轴’)
plt.ylabel(‘Y轴’)

plt.plot(X,Y,label=’aa’,c=’red’) #折线图

#直方图

plt.bar(X,Y,label=’bb’,color=’pink’)

#散点图

plt.scatter(X,Y,label=’cc’,c=’blue’,marker=’o’)

#显示图例

plt.legend()

#饼图

labels = [‘期中笔试’,’期中基试’,’期末笔试’,’期末基试’]
data = [30,50,60,70]
explode=[0.1,0.1,0.1,0] #板块间隔
plt.pie(data,labels=labels,autopct=’%1.1f%%’,shadow=True,startangle=100,explode=explode) #autopct = ‘%1.1f%%’
plt.axis(‘equal’) #长宽相同
plt.title(‘死胖子’)
plt.savefig(‘score.png’) #保存图片
plt.show()
